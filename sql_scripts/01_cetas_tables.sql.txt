CREATE EXTERNAL TABLE dbo.dim_rider
WITH (
    LOCATION = 'gold/dim_rider',
    DATA_SOURCE = BikeShareDataSource,
    FILE_FORMAT = SynapseDelimitedTextFormat
)
AS
SELECT
    rider_id,
    first_name,
    last_name,
    birthday,
    is_member AS member_type
FROM dbo.staging_rider;

CREATE EXTERNAL TABLE dbo.dim_station
WITH (
    LOCATION = 'gold/dim_station',
    DATA_SOURCE = BikeShareDataSource,
    FILE_FORMAT = SynapseDelimitedTextFormat
)
AS
SELECT
    station_id,
    name AS station_name,
    latitude,
    longitude
FROM dbo.staging_station;

CREATE EXTERNAL TABLE dbo.dim_date
WITH (
    LOCATION = 'gold/dim_date',
    DATA_SOURCE = BikeShareDataSource,
    FILE_FORMAT = SynapseDelimitedTextFormat
)
AS
SELECT DISTINCT
    CAST(start_at AS DATE) AS full_date,
    DAY(CAST(start_at AS DATE)) AS day,
    MONTH(CAST(start_at AS DATE)) AS month,
    DATEPART(QUARTER, CAST(start_at AS DATE)) AS quarter,
    YEAR(CAST(start_at AS DATE)) AS year,
    DATENAME(WEEKDAY, CAST(start_at AS DATE)) AS weekday
FROM dbo.staging_trip;

CREATE EXTERNAL TABLE dbo.fact_payment
WITH (
    LOCATION = 'gold/fact_payment',
    DATA_SOURCE = BikeShareDataSource,
    FILE_FORMAT = SynapseDelimitedTextFormat
)
AS
SELECT
    payment_id,
    rider_id,
    CAST(payment_date AS DATE) AS payment_date,
    amount
FROM dbo.staging_payment;

CREATE EXTERNAL TABLE dbo.fact_trip
WITH (
    LOCATION = 'gold/fact_trip',
    DATA_SOURCE = BikeShareDataSource,
    FILE_FORMAT = SynapseDelimitedTextFormat
)
AS
SELECT
    t.trip_id,
    t.rider_id,
    t.start_station_id,
    t.end_station_id,
    CAST(t.start_at AS DATE) AS date_id,
    DATEDIFF(
        MINUTE,
        CAST(t.start_at AS DATETIME2),
        CAST(t.ended_at AS DATETIME2)
    ) AS duration_minutes,
    DATEDIFF(
        YEAR,
        CAST(r.birthday AS DATE),
        CAST(t.start_at AS DATE)
    ) AS rider_age
FROM dbo.staging_trip t
JOIN dbo.staging_rider r
ON t.rider_id = r.rider_id;
